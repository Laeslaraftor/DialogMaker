	public class Array<T> : IEnumerable<T>
	{
		public number Length { get; }
		public T this[number index]
		{
			get => GetValue(index);
			set => SetValue(index, value);
		}
		
		private extern func SetValue(number index, T value);
		private extern T GetValue(number index);
		
		public IEnumerator<T> GetEnumerator()
		{
			return new Enumerator(this);
		}
		
		public static func Copy(Array<T> source, number sourceIndex, Array<T> destination, number destinationIndex, number count)
		{
			for (number i = 0; i < count; i++)
			{
				destination[destinationIndex + i] = source[sourceIndex + i];
			}
		}
		
		private class Enumerator : IEnumerator<T>
		{
			public Enumerator(T[] items)
			{
				_items = items;
			}
			
			public T Current { get; private set; }
		
			private readonly T[] _items;
			private number _currentIndex;
		
			public bool MoveNext()
			{
				if (_items.Length == 0 ||
					_currentIndex + 1 >= _items.Length)
				{
					return false;
				}
				
				_currentIndex++;
				Current = _items[_currentIndex];
				
				return true;
			}
			public func Reset()
			{
				_currentIndex = -1;
				Current = null;
			}
		}
	}

	public class List<T> : IList<T>
	{
		public List()
		{
			_items = new T[4];
			Capacity = 4;
		}
		
		public number Count { get; private set; }
		public number Capacity { get; private set; }
		public T this[number index]
		{
			get => _items[index];
			set => _items[index] = value;
		}		
		
		private T[] _items;
		
		public number IndexOf(T item)
		{
			for (number i = 0; i < Count; i++)
			{
				if (_items[i] == item)
				{
					return i;
				}
			}
			
			return -1;
		}
		public bool Contains(T item) => IndexOf(item) != -1;
		public func Insert(number index, T item)
		{
			if (index >= Count)
			{
				Add(item);
				return;
			}
			
			ExpandIfNeed();
			bool isAdded = false;
			
			for (number i = Count - 1; i >= 0; i--)
			{
				_items[i + 1] = _items[i];
				
				if (i == index)
				{
					_items[i] = item;
					break;
				}
			}
			
			Count++;
		}
		public func Add(T item)
		{
			ExpandIfNeed();
			
			_items[Count] = item;
			Count++;
		}
		public bool Remove(T item)
		{
			if (Count == 0)
			{
				return false;
			}
			
			bool isRemoved = false;
			
			for (number i = 0; i < Count; i++)
			{
				T currentItem = _items[i];
				
				if (isRemoved)
				{
					_items[i - 1] = currentItem;
					continue;
				}
				if (currentItem == item)
				{
					isRemoved = true;
				}
			}
			
			if (isRemoved)
			{
				Count--;
			}
			
			return isRemoved;
		}
		public func Clear()
		{
			if (Count == 0)
			{
				return;
			}
			
			for (number i = 0; i < Count; i++)
			{
				_items[i] = null;
			}
			
			Count = 0;
		}
		
		private func ExpandIfNeed()
		{
			if (Count + 1 > Capacity)
			{
				Expand();
			}
		}
		private func Expand()
		{
			Capacity *= 2;
			T[] newItems = new T[Capacity];
			Array<T>.Copy(_items, 0, newItems, 0, _items.Length);			
			_items = newItems;
		}
	}

string repeat(string str, number count)

Color getTextColor(string text)
{
	if (text.Length < 3)
	{
		return Color.Black;
	}
	
	number[] channels = new number[3];
	number charsPerChannel = text.Length / 3;	

	for (number channel = 0; channel < channels.Length; channel++) 
	{
		for (number i = channel; i < charsPerChannel; i++)
		{
			channels[channel] += text[i + channel * charsPerChannel];
		}
		
		channels[channel] /= charsPerChannel;
	}
	
	return new Color(channels[0], channels[1], channels[2]);
}
