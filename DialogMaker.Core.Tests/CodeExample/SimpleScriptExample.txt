			public Enumerator(T[] items)
			{
				_items = items;
			}
			
			public T Current { get; private set; }
		
			private readonly T[] _items;
			private int _currentIndex;
		
			public bool MoveNext()
			{
				if (_items.Length == 0 ||
					_currentIndex + 1 >= _items.Length)
				{
					return false;
				}
				
				_currentIndex++;
				Current = _items[_currentIndex];
				
				return true;
			}

				_currentIndex = -1;
				Current = null;
			}

	public class List<T> : IList<T>
	{
		public List()
		{
			_items = new T[4];
			Capacity = 4;
		}
		
		public int Count { get; private set; }
		public int Capacity { get; private set; }
		public T this[int index]
		{
			get => _items[index];
			set => _items[index] = value;
		}		
		
		private T[] _items;
		
		public int IndexOf(T item)
		{
			for (int i = 0; i < Count; i++)
			{
				if (_items[i] == item)
				{
					return i;
				}
			}
			
			return -1;
		}
		public bool Contains(T item) => IndexOf(item) != -1;
		public void Insert(int index, T item)
		{
			if (index >= Count)
			{
				Add(item);
				return;
			}
			
			ExpandIfNeed();
			bool isAdded = false;
			
			for (int i = Count - 1; i >= 0; i--)
			{
				_items[i + 1] = _items[i];
				
				if (i == index)
				{
					_items[i] = item;
					break;
				}
			}
			
			Count++;
		}
		public void Add(T item)
		{
			ExpandIfNeed();
			
			_items[Count] = item;
			Count++;
		}
		public bool Remove(T item)
		{
			if (Count == 0)
			{
				return false;
			}
			
			bool isRemoved = false;
			
			for (int i = 0; i < Count; i++)
			{
				T currentItem = _items[i];
				
				if (isRemoved)
				{
					_items[i - 1] = currentItem;
					continue;
				}
				if (currentItem == item)
				{
					isRemoved = true;
				}
			}
			
			if (isRemoved)
			{
				Count--;
			}
			
			return isRemoved;
		}
		public void Clear()
		{
			if (Count == 0)
			{
				return;
			}
			
			for (int i = 0; i < Count; i++)
			{
				_items[i] = null;
			}
			
			Count = 0;
		}
		
		public IEnumerator<T> GetEnumerator()
		{
			return new ListEnumerator<T>(this);
		}
		
		private void ExpandIfNeed()
		{
			if (Count + 1 > Capacity)
			{
				Expand();
			}
		}
		private void Expand()
		{
			Capacity *= 2;
			T[] newItems = new T[Capacity];
			Array<T>.Copy(_items, 0, newItems, 0, _items.Length);			
			_items = newItems;
		}
	}
	public class ListEnumerator<T> : IEnumerator<T>
	{
		public Enumerator(IList<T> list)
		{
			_list = list;
		}
		
		public T Current { get; private set; }
		
		private readonly IList<T> _list;
		private int _currentIndex;
		
		public bool MoveNext()
		{
			if (_list.Count == 0 ||
				_currentIndex + 1 >= _list.Count)
			{
				return false;
			}
			
			_currentIndex++;
			Current = _list[_currentIndex];
			
			return true;
		}
		public void Reset()
		{
			_currentIndex = -1;
			Current = null;
		}
	}

Color getTextColor(string text)
{
	if (text.Length < 3)
	{
		return Color.Black;
	}
	
	int[] channels = new int[3];
	int charsPerChannel = text.Length / 3;	

	for (int channel = 0; channel < channels.Length; channel++) 
	{
		for (int i = channel; i < charsPerChannel; i++)
		{
			channels[channel] += text[i + channel * charsPerChannel];
		}
		
		channels[channel] /= charsPerChannel;
	}
	
	return new Color(channels[0], channels[1], channels[2]);
}
