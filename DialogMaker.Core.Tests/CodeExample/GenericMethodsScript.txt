namespace System
{
	public sealed struct Byte {}
	public sealed struct Int32 {}
	public sealed struct Decimal {}
	public sealed struct Single {}
	public sealed struct Double {}
	public sealed struct Boolean {}
	public sealed class String {}
	public sealed class Object {}
	public sealed struct Void {}
	public class Exception {}
	
	namespace Collections
	{
		namespace Generic
		{
			public interface IList<T>
			{
			}
		}
	}
}

public class MegaClass
{	
	public string Name { get; }

	public TCast Cast<TCast, TItem>()
		where TCast : IList<TItem>
	{
		return null;
	}
	public static T CreateInstance<T>()
		where T : MegaClass, new()
	{
		return new();
	}
}

void GetHudoeName()
{
	var instance = MegaClass.CreateInstance<MegaClass>();
	
	if (instance == null)
	{
		throw new Exception();
	}
	else 
	{
		GetHudoeName();
	}
}
