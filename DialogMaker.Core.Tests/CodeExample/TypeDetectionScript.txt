namespace System
{
	public class Array<T>
	{
		public T this[int index]
		{
			get => GetValue(index);
			set => SetValue(index, value);
		}
		
		private extern T GetValue(int index);
		private extern void SetValue(int index, T value);
	}
	public class String
	{
		public char this[int index]
		{
			get => GetValue(index);
			set => SetValue(index, value);
		}
		
		public char GetFirstSymbol() => this[0];
		
		public static string operator +(string l, object r) => l + r.ToString();
		public static string operator +(object l, string r) => l.ToString() + r;
		public static string operator +(string l, string r) => l;
		
		private extern char GetValue(int index);
		private extern void SetValue(int index, char value);
	}
	public class Object
	{
		public extern Type GetType();
		public virtual string ToString() => GetType().Name;
	}
	public struct Char {}
	public struct Boolean {}
	public struct Int32 {}
	public struct Void {}
	
	public class Type
	{
		public string Name { get; }
	}
}

public class Group
{
	public virtual Character this[int index]
	{
		get => _characters[index];
		set => _characters[index] = value;
	}
	
	private readonly Character[] _characters;
}
public class PlayerGroup : Group
{
	public override Character this[int index]
	{
		get => base.this[index];
		set => base.this[index] = value;
	}
}
public class Character
{
	public string Name { get; set; }
	
	public override string ToString()
	{
		if (Name != null)
		{
			return Name;
		}
		
		return base.ToString();
	}
	public void PrintName()
	{
		Name[10] = 'c';
	}
	protected Character GetPath(int depth)
	{
		return this;
	}
}
